--* DEL MODULO CST006.                    *--
--* CONSULTOR3 - 30/05/2016.              *--
--*---------------------------------------*--

RETURN number AS

vcount number;

begin

select count(*) into vcount 
from CST006
WHERE ORDEN || ANOCAL || MESCAL || ORDINI || ORDFIN  like '%'||pbusqueda||'%'

end;
